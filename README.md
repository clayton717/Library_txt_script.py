# Track all unique entries and where they first appeared
seen = {}
duplicates = []

# Option: Change to True to alphabetize entries within each category
SORT_ENTRIES = True

SORT_CATEGORIES = True

new_media_lib = OrderedDict()

# Get categories in desired order
categories = list(MEDIA_LIB.keys())
if SORT_CATEGORIES:
    categories = sorted(categories)

for cat in categories:
    deduped = []
    for entry in MEDIA_LIB[cat]:
        if entry not in seen:
            seen[entry] = cat
            deduped.append(entry)
        else:
            duplicates.append((entry, cat, seen[entry]))
    if SORT_ENTRIES:
        deduped = sorted(deduped, key=lambda x: (x[0].lower(), x[1]))
    new_media_lib[cat] = deduped

MEDIA_LIB = new_media_lib

def hash_sha256(data):
    if isinstance(data, str):
        data = data.encode("utf-8")
    h = hashlib.sha256()
    h.update(data)
    return h.hexdigest()

def fetch_about_or_wiki(url, site_name):
    try:
        about_url = url.rstrip('/') + "/about"
        r = requests.get(about_url, timeout=10, headers={"User-Agent": "Mozilla/5.0"})
        if r.status_code == 200:
            soup = BeautifulSoup(r.text, "html.parser")
            paragraphs = soup.find_all('p')
            if paragraphs:
                return "\n".join(p.get_text() for p in paragraphs[:3])
    except Exception:
        pass
    try:
        wiki_api = f"https://en.wikipedia.org/api/rest_v1/page/summary/{site_name.replace(' ', '_')}"
        r = requests.get(wiki_api, timeout=10)
        if r.status_code == 200:
            data = r.json()
            return "Wikipedia: " + data.get('extract', '')
    except Exception:
        pass
    return ""

def fetch_web_content(url, chars=4000):
    try:
        r = requests.get(url, timeout=10, headers={"User-Agent": "Mozilla/5.0"})
        soup = BeautifulSoup(r.text, "html.parser")
        paragraphs = soup.find_all('p')
        text = "\n".join(p.get_text() for p in paragraphs) if paragraphs else soup.get_text(separator="\n", strip=True)
        return text[:chars]
    except Exception as e:
        return f"[Error fetching {url}: {e}]"

def log_event(event, data):
    log = []
    if os.path.exists(LOG_FILE):
        with open(LOG_FILE, 'r', encoding='utf-8') as f:
            try:
                log = json.load(f)
            except Exception:
                log = []
    log.append({"event": event, "data": data, "time": time.strftime("%Y-%m-%d %H:%M:%S")})
    with open(LOG_FILE, 'w', encoding='utf-8') as f:
        json.dump(log, f, indent=2)

def build_super_corpus():
    entries = []
    hashes = set()
    print("=== Web/Library Import ===")
    for category, resources in MEDIA_LIB.items():
        entries.append(f"\n\n=== Category: {category} ===\n\n")
        for desc, url, summary in resources:
            site_name = desc.strip()
            about_or_wiki = fetch_about_or_wiki(url, site_name)
            web_content = fetch_web_content(url, chars=CHARS_PER_SITE)
            entry = f"{desc}\n{summary}\n{url}\n{about_or_wiki}\n{web_content}\n"
            h = hash_sha256(entry)
            if h not in hashes:
                entries.append(entry)
                hashes.add(h)
                log_event("import_resource", {"category": category, "desc": desc, "url": url})
            print(f"Added: {desc}")
            time.sleep(1)
    print("=== Deduplicating and Saving ===")
    with open(OUTPUT_FILE, 'w', encoding='utf-8') as f:
        for entry in entries:
            f.write(entry + "\n")
    print(f"\nSuper corpus built in {OUTPUT_FILE} ({len(hashes)} unique entries)")
    print(f"SHA256: {hash_sha256(''.join(entries))}")
    log_event("corpus_built", {"entries": len(hashes), "hash": hash_sha256(''.join(entries))})

if __name__ == "__main__":
    build_super_corpus()
